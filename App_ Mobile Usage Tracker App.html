              hasNewAlert = true;

            updatedApps[i] = { ...app, timeSpent: newTime };
          }

            variant={activeTab === "dashboard" ? "default" : "ghost"}
            className={`flex-1 rounded-md ${activeTab === "dashboard" ? "bg-blue-500 text-white hover:bg-blue-600" : ""}`}
            onClick={() => setActiveTab("dashboard")}
          >
            <BarChartIcon className="mr-2 h-4 w-4" />
            Dashboard
          </Button>
          <Button
            variant={activeTab === "settings" ? "default" : "ghost"}
            className={`flex-1 rounded-md ${activeTab === "settings" ? "bg-blue-500 text-white hover:bg-blue-600" : ""}`}
            onClick={() => setActiveTab("settings")}
          >
            <SettingsIcon className="mr-2 h-4 w-4" />
            Settings
          </Button>
        </div>

        {/* Dashboard View */}
        {activeTab === "dashboard" && (
          <div className="space-y-6">
            {/* Summary Card */}
            <Card className="shadow-sm">
              <CardHeader>
                <CardTitle className="flex items-center">
                  <ClockIcon className="mr-2 h-5 w-5" />
                  Today's Usage Summary
                </CardTitle>
              </CardHeader>
              <CardContent>
                <div className="grid grid-cols-1 gap-4 md:grid-cols-3">
                  <div className="rounded-lg bg-blue-50 p-4 text-center">
                    <p className="text-sm text-gray-600">Total Time</p>
                    <p className="text-2xl font-bold text-blue-600">{formatTime(totalUsage)}</p>
                  </div>
                  <div className="rounded-lg bg-green-50 p-4 text-center">
                    <p className="text-sm text-gray-600">Apps Tracked</p>
                    <p className="text-2xl font-bold text-green-600">{activeApps.length}</p>
                  </div>
                  <div className="rounded-lg bg-orange-50 p-4 text-center">
                    <p className="text-sm text-gray-600">Limits Exceeded</p>
                    <p className="text-2xl font-bold text-orange-600">{exceededApps.length}</p>
                  </div>
                </div>
              </CardContent>
            </Card>

            {/* Apps List */}
            <Card className="shadow-sm">
              <CardHeader>
                <CardTitle>Social Media Apps</CardTitle>
                <CardDescription>Track your usage across different platforms</CardDescription>
              </CardHeader>
              <CardContent>
                {apps.length === 0 ? (
                  <div className="py-8 text-center">
                    <p className="text-gray-500">No apps added yet. Go to Settings to add apps.</p>
                  </div>
                ) : (
                  <div className="space-y-4">
                    {apps.map((app) => {
                      const progress = calculateProgress(app.timeSpent, app.dailyLimit);
                      const progressColor = getProgressColor(progress);
                      
                      return (
                        <div 
                          key={app.id} 
                          className="flex flex-col rounded-lg border p-4 hover:bg-gray-50 md:flex-row md:items-center"
                        >
                          <div className="flex items-center">
                            <span className="mr-3 text-2xl">{app.icon}</span>
                            <div>
                              <h3 className="font-medium">{app.name}</h3>
                              <div className="flex items-center text-sm text-gray-500">
                                <ClockIcon className="mr-1 h-3 w-3" />
                                <span>{formatTime(app.timeSpent)} / {formatTime(app.dailyLimit)}</span>
                              </div>
                            </div>
                          </div>
                          
                          <div className="mt-3 flex flex-1 flex-col md:ml-4 md:mt-0">
                            <div className="mb-1 flex items-center justify-between">
                              <span className="text-xs text-gray-500">{progress}%</span>
                              <span className="text-xs text-gray-500">
                                {app.dailyLimit - app.timeSpent > 0 
                                  ? `${formatTime(app.dailyLimit - app.timeSpent)} remaining` 
                                  : "Limit reached"}
                              </span>
                            </div>
                            <div className="h-2 w-full rounded-full bg-gray-200">
                              <div 
                                className={`h-full rounded-full ${progressColor}`} 
                                style={{ width: `${progress}%` }}
                              ></div>
                            </div>
                          </div>
                          
                          <div className="mt-3 flex items-center md:ml-4 md:mt-0">
                            {app.timeSpent >= app.dailyLimit ? (
                              <Badge variant="destructive" className="mr-3">
                                <AlertIcon className="mr-1 h-3 w-3" />
                                Limit Reached
                              </Badge>
                            ) : (
                              <Badge variant="secondary" className="mr-3">
                                {progress}%
                              </Badge>
                            )}
                            
                            <Button
                              variant="ghost"
                              onClick={() => toggleTracking(app.id)}
                              className="mr-2"
                            >
                              {app.isTracking ? (
                                <PauseIcon className="h-4 w-4 text-gray-600" />
                              ) : (
                                <PlayIcon className="h-4 w-4 text-gray-600" />
                              )}
                            </Button>
                            
                            <Button
                              variant={app.isActive ? "outline" : "secondary"}
                              onClick={() => toggleAppStatus(app.id)}
                            >
                              {app.isActive ? "Active" : "Paused"}
                            </Button>
                          </div>
                        </div>
                      );
                    })}
                  </div>
                )}
              </CardContent>
            </Card>
          </div>
        )}

        {/* Settings View */}
        {activeTab === "settings" && (
          <div className="space-y-6">
            {/* Add New App */}
            <Card className="shadow-sm">
              <CardHeader>
                <CardTitle className="flex items-center">
                  <PlusIcon className="mr-2 h-5 w-5" />
                  Add New App
                </CardTitle>
              </CardHeader>
              <CardContent>
                <div className="grid grid-cols-1 gap-4 md:grid-cols-4">
                  <div className="md:col-span-2">
                    <label className="mb-1 block text-sm font-medium text-gray-700">App Name</label>
                    <Input
                      value={newAppName}
                      onChange={(e) => setNewAppName(e.target.value)}
                      placeholder="e.g., YouTube, Snapchat"
                    />
                  </div>
                  <div>
                    <label className="mb-1 block text-sm font-medium text-gray-700">Daily Limit (min)</label>
                    <Input
                      type="number"
                      min="1"
                      value={newAppLimit}
                      onChange={(e) => setNewAppLimit(Number(e.target.value))}
                    />
                  </div>
                  <div className="flex items-end">
                    <Button onClick={addApp} className="w-full">
                      Add App
                    </Button>
                  </div>
                </div>
              </CardContent>
            </Card>

            {/* Manage Apps */}
            <Card className="shadow-sm">
              <CardHeader>
                <CardTitle>Manage Apps</CardTitle>
                <CardDescription>Configure tracking settings for each app</CardDescription>
              </CardHeader>
              <CardContent>
                {apps.length === 0 ? (
                  <div className="py-8 text-center">
                    <p className="text-gray-500">No apps added yet.</p>
                  </div>
                ) : (
                  <div className="space-y-4">
                    {apps.map((app) => (
                      <div 
                        key={app.id} 
                        className="flex flex-col items-start justify-between gap-4 rounded-lg border p-4 md:flex-row md:items-center"
                      >
                        <div className="flex items-center">
                          <span className="mr-3 text-2xl">{app.icon}</span>
                          <div>
                            <h3 className="font-medium">{app.name}</h3>
                            <p className="text-sm text-gray-500">
                              Current limit: {formatTime(app.dailyLimit)}
                            </p>
                          </div>
                        </div>
                        
                        <div className="flex w-full items-center gap-2 md:w-auto">
                          <Input
                            type="number"
                            min="1"
                            value={app.dailyLimit}
                            onChange={(e) => updateAppLimit(app.id, Number(e.target.value))}
                            className="w-24"
                          />
                          <span className="text-sm text-gray-500">minutes</span>
                          <Button
                            variant="ghost"
                            onClick={() => removeApp(app.id)}
                            className="text-red-500 hover:text-red-700"
                          >
                            <TrashIcon className="h-4 w-4" />
                          </Button>
                        </div>
                      </div>
                    ))}
                  </div>
                )}
              </CardContent>
            </Card>

            {/* Simulation Settings */}
            <Card className="shadow-sm">
              <CardHeader>
                <CardTitle className="flex items-center">
                  <SettingsIcon className="mr-2 h-5 w-5" />
                  Simulation Settings
                </CardTitle>
              </CardHeader>
              <CardContent>
                <div className="flex flex-col gap-4 md:flex-row md:items-center">
                  <div>
                    <label className="mb-1 block text-sm font-medium text-gray-700">Simulation Speed</label>
                    <div className="flex items-center gap-2">
                      <input
                        type="range"
                        min="100"
                        max="5000"
                        step="100"
                        value={simulationSpeed}
                        onChange={(e) => setSimulationSpeed(Number(e.target.value))}
                        className="w-48"
                      />
                      <span className="text-sm text-gray-600">
                        {simulationSpeed}ms per minute
                      </span>
                    </div>
                  </div>
                  <div className="text-sm text-gray-500">
                    <p>Adjust how fast time passes in the simulation</p>
                  </div>
                </div>
              </CardContent>
            </Card>

            {/* Usage Tips */}
            <Card className="shadow-sm">
              <CardHeader>
                <CardTitle className="flex items-center">
                  <CheckIcon className="mr-2 h-5 w-5 text-green-500" />
                  Usage Tips
                </CardTitle>
              </CardHeader>
              <CardContent>
                <ul className="list-inside list-disc space-y-2 text-gray-600">
                  <li>Set realistic daily limits based on your schedule</li>
                  <li>Start with higher limits and gradually reduce them</li>
                  <li>Use the pause feature when you need to use social media for work</li>
                  <li>Review your weekly usage to identify patterns</li>
                  <li>Consider turning off notifications for apps you want to limit</li>
                </ul>
              </CardContent>
            </Card>
          </div>
        )}

        {/* Footer */}
        <footer className="mt-8 text-center text-sm text-gray-500">
          <p>Your data is stored locally on your device</p>
        </footer>
      </div>
    </div>
  );
}
